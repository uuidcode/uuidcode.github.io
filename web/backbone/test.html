    AppView2 = Backbone.View.extend({
        tagName: 'span',
        class: 'myClass',
        id: 'myId',
        initialize: function () {
            this.render();
        },
        render: function () {
            this.$el.html('Hello, World')
                .appendTo($('#view2'));
        }
    });

    var appView2 = new AppView2();
