Methods

rolling(rollingDirection, width, height, [viewingItemCount])
 - init method
 - rollingDirection String ("up", "down", "right", "left"), String
 - the width of a rolling item, Number
 - the height of a rolling item, Number
 - the count of the viewing item, Number, optional
 - return jQuery 

addRollingItem(html)
 - add rolling item
 - html String
 - return jQuery
 
 startRolling(rollingTime, viewingTime, rollingAnimationFrame)
 - start rolling
 - rollingTime Number
 - viewingTime Number
 - rollingAnimationFrame Number
 - return jQuery
 
 stopRolling()
 - stop rolling
 - return jQuery
 
 resumeRolling()
 - resume rolling
 - return jQuery
 
 reverseRolling()
	from left-to-right -> from right-to-left
	from right-to-left -> left-to-right
	from bottom-to-top -> top-to-bottom
	from top-to-bottom -> bottom-to-top
 - return jQuery
